# 火山引擎嵌入模型类（使用官方SDK）
class VolcEngineEmbeddings:
    """
    自定义嵌入模型类，用于将文本转换为向量表示
    使用火山引擎的嵌入模型服务
    """

    def __init__(self, api_key, model_name = "doubao-embedding-text-240715"):
        """
        初始化火山引擎嵌入模型

        参数:
        - api_key: 火山引擎API密钥
        - model_name: 使用的模型名称"
        """
        self.api_key = api_key
        self.model_name = model_name
        self.client = Ark(api_key=api_key)  # 创建火山引擎客户端实例

    def embed_documents(self, texts):
        """
        为多个文档生成嵌入向量

        参数:
        - texts: 文本列表，每个元素是一个字符串

        返回:
        - 嵌入向量列表，每个元素是一个浮点数列表表示文本的向量
        """
        try:
            embeddings = []  # 存储所有文本的嵌入向量
            for text in texts:
                # 调用火山引擎API生成嵌入向量
                response = self.client.embeddings.create(
                    model=self.model_name,  # 指定模型
                    input=text)  # 输入文本，单个字符串


                # 检查响应是否有效
                if response and hasattr(response, 'data') and response.data:
                    embedding_data = response.data[0].embedding  # 获取嵌入数据
                    # 确保返回的是列表格式
                    if isinstance(embedding_data, list):
                        embeddings.append(embedding_data)  # 直接添加列表
                    else:
                        # 如果是其他格式，尝试转换为列表
                        try:
                            embeddings.append(list(embedding_data))
                        except:
                            st.error(f"嵌入格式转换失败: {type(embedding_data)}")
                            return []  # 转换失败返回空列表
                else:
                    st.error(f"嵌入生成失败，响应: {response}")
                    return []  # API调用失败返回空列表

            return embeddings  # 返回所有嵌入向量

        except Exception as e:
            # 异常处理，显示错误信息和堆栈跟踪
            st.error(f"嵌入请求失败: {str(e)}")
            import traceback
            st.error(traceback.format_exc())
            return []  # 异常时返回空列表

    def embed_query(self, text):
        """
        为查询文本生成嵌入向量（与文档嵌入使用相同方法）

        参数:
        - text: 查询文本字符串

        返回:
        - 嵌入向量，浮点数列表
        """
        try:
            # 调用火山引擎API生成查询嵌入
            response = self.client.embeddings.create(
                model=self.model_name,
                input=text  # 查询文本
            )

            # 处理响应
            if response and hasattr(response, 'data') and response.data:
                embedding_data = response.data[0].embedding
                if isinstance(embedding_data, list):
                    return embedding_data  # 直接返回列表
                else:
                    try:
                        return list(embedding_data)  # 尝试转换为列表
                    except:
                        st.error("查询嵌入格式转换失败")
                        # 转换失败时使用文档嵌入方法作为备选
                        return self.embed_documents([text])[0]
            else:
                # API响应无效时使用文档嵌入方法作为备选
                return self.embed_documents([text])[0]

        except Exception as e:
            # 异常处理
            st.error(f"查询嵌入失败: {str(e)}")
            # 异常时使用文档嵌入方法作为备选
            return self.embed_documents([text])[0]


# DeepSeek模型自定义类
class DeepSeekLLM:
    """
    自定义DeepSeek语言模型类
    """

    def __init__(self, api_key, api_base, model_name = "deepseek-chat", t = 0.1):
        """
        初始化DeepSeek模型

        参数:
        - api_key: DeepSeek API密钥
        - api_base: API基础URL
        - model_name: 模型名称，默认为"deepseek-chat"
        - t: 生成温度，控制输出的随机性
        """
        self.api_key = api_key
        self.api_base = api_base
        self.model_name = model_name
        self.t = t

    def invoke(self, prompt):
        """
        调用DeepSeek API生成文本

        参数:
        - prompt: 输入的提示文本

        返回:
        - 模型生成的回答文本
        """
        try:
            # 设置HTTP请求头
            headers = {
                "Content-Type": "application/json",  # JSON格式
                "Authorization": f"Bearer {self.api_key}"  # Bearer token认证
            }

            # 构建请求负载
            payload = {
                "model": self.model_name,  # 模型名称
                "messages": [
                    {
                        "role": "user",  # 用户角色
                        "content": prompt  # 提示内容
                    }
                ],
                "temperature": self.t,  # 温度参数
                "stream": False  # 非流式输出
            }

            # 发送POST请求到DeepSeek API
            # post是发包(发到DeepSeek服务器上)
            response = requests.post(
                f"{self.api_base}/chat/completions",  # API端点
                headers=headers,  # 请求头
                json=payload,  # 请求体
                timeout=60  # 超时时间60秒(服务器超过60秒无响应，退出不再等待)
            )

            # 检查响应状态码，根据状态码判断处理结果
            # 响应码为200，表示服务器处理成功
            if response.status_code == 200:
                # 解析JSON响应
                result = response.json()
                # 返回生成的文本内容
                return result['choices'][0]['message']['content']
            else:
                # API调用失败，显示错误信息
                st.error(f"API调用失败: {response.status_code} - {response.text}")
                return f"抱歉，生成答案时出错: {response.text}"

        except Exception as e:
            # 异常处理
            st.error(f"调用DeepSeek API时出错: {str(e)}")
            return f"抱歉，生成答案时出错: {str(e)}"

# 2. 定义Main函数作为程序的入口
def main():
    st.header("Ask for PDF")  # 主页面标题

    # 检查API密钥是否设置（从global_var导入）
    # API_KEY为空，意味着无法与大模型对话，直接退出
    if not API_KEY:
        st.error("请设置火山引擎API密钥")  # 显示错误信息
        return  # 如果没有API密钥，直接返回

    # 在侧边栏添加测试嵌入功能的按钮
    embedding_test()

    # 创建火山引擎嵌入实例，用于后续的文本嵌入
    embeddings = VolcEngineEmbeddings(api_key=API_KEY)

    # 3. 上传PDF文件
    # 创建文件上传器组件，只接受PDF格式
    pdf = st.file_uploader("请上传PDF", type="pdf")
    if pdf is not None:  # 如果用户上传了文件
        st.write(f"已上传文件: {pdf.name}")  # 显示文件名

        # 读取PDF文件内容
        try:
            pdf_reader = PdfReader(pdf)  # 创建PDF阅读器
            text = ""  # 存储提取的文本
            for page in pdf_reader.pages:  # 遍历每一页
                page_text = page.extract_text()  # 提取页面文本
                if page_text:
                    text += page_text + "\n"  # 拼接文本

            # 检查是否成功提取文本
            if not text.strip():
                st.error("无法从PDF中提取文本，请检查PDF文件")  # 错误提示
                return  # 返回，不再继续执行

            # 显示提取的文本（前1000字符）
            with st.expander("查看提取的文本"):
                st.text(text[:1000] + "..." if len(text) > 1000 else text)

        except Exception as e:
            st.error(f"读取PDF文件时出错: {str(e)}")  # 异常处理
            return  # 返回，不再继续执行

        # 4. 对文本进行切割
        # 创建递归字符文本分割器
        text_splitter = RecursiveCharacterTextSplitter(
            chunk_size=500,  # 每个块的大小（字符数）
            chunk_overlap=50,  # 块之间的重叠字符数
            length_function=len  # 长度计算函数
        )
        # 分割文本为多个块
        chunks = text_splitter.split_text(text=text)
        # 显示分割结果
        st.write(f"文本被分割成 {len(chunks)} 个片段")

        # 5. 将切割后的文件块embedding到向量数据库Chroma中
        try:
            # 先测试少量文本的嵌入功能
            st.info("正在测试嵌入功能...")  # 信息提示
            test_chunks = chunks[:2]  # 只测试前两个片段
            test_embeddings_result = embeddings.embed_documents(test_chunks)  # 测试嵌入

            # 检查测试结果
            if not test_embeddings_result:
                st.error("嵌入功能测试失败，无法创建向量数据库")  # 错误提示
                return  # 返回，不再继续执行

            st.success("嵌入功能测试成功，正在创建向量数据库...")  # 成功提示

            # 创建向量数据库
            # 将文本块转换为向量并存储到Chroma数据库中
            vectorStore = Chroma.from_texts(
                texts=chunks,  # 文本块列表
                embedding=embeddings,  # 嵌入模型
                metadatas=[{"source": f"chunk_{i}"} for i in range(len(chunks))],  # 元数据
                persist_directory="./chroma_db"  # 持久化目录
            )

            st.success("向量数据库创建成功！")  # 成功提示

            # 6. 用户输入，搜索向量数据库，返回结果
            # 创建文本输入框供用户输入问题
            query = st.text_input("请输入与PDF相关的问题：")
            if query:  # 如果用户输入了问题
                with st.spinner("正在搜索和生成答案..."):  # 显示加载 spinner
                    # 创建检索器，用于从向量数据库中检索相关文档
                    retriever = vectorStore.as_retriever(search_kwargs={"k": 3})  # 返回最相关的3个文档

                    # 检索相关文档，query为用户在输入框中输入的问题
                    relevant_docs = retriever.invoke(query)  # 执行检索

                    # 构建上下文，将检索到的文档内容拼接成一个字符串
                    # 如果检索到多个文档，将多个文档拼接成一个字符串
                    # 每个文档之间一个空行进行分隔
                    context = "\n\n".join([doc.page_content for doc in relevant_docs])

                    # 使用自定义的DeepSeek模型
                    llm = DeepSeekLLM(
                        api_key=ds_key,  # DeepSeek API密钥（从global_var导入）
                        api_base=ds_url,  # DeepSeek API地址（从global_var导入）
                        t=0.1  # 温度参数
                    )

                    # 构建提示词模板
                    prompt_template = """
                    请根据以下上下文信息回答问题。如果上下文信息不足以回答问题，请回答”知识库中未找到相关信息“。

                    上下文信息：
                    {context}

                    问题：
                    {question}

                    请提供准确、简洁的回答：
                    """

                    # 格式化提示词，插入上下文和问题
                    # context是从向量数据库中检索到的文档片段

                                                    question=query)

                        for i, doc in enumerate(relevant_docs):
                            # 显示每个参考文档的前200个字符
                            st.write(f"文档 {i + 1}: {doc.page_content[:200]}...")
